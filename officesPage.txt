@FindBy (xpath ="//main[@id='container']/section/div")
    private List<WebElement> totalRegionsPlaceHolder;


  public int ReturnTotalNumberOfOfficesDisplay()
    {
        int totalOffice = totalRegionsPlaceHolder.size();
        System.out.println("Total number of Regions Displayed "+ totalOffice);
        int allOfficesDisplayed = 0;
        for(int i = 0; i < totalOffice; i++)
        {
            WebElement regionText = totalRegionsPlaceHolder.get(i);
            String regionText1 = regionText.findElement(By.tagName("h2")).getText();
            WebElement parentRegion = totalRegionsPlaceHolder.get(i).findElement(By.tagName("ul"));
            int totalOfficesInRegion = parentRegion.findElements(By.tagName("li")).size();
            System.out.println(regionText1+" has a total number of "+totalOfficesInRegion+" offices in it region.");
            allOfficesDisplayed = allOfficesDisplayed + totalOfficesInRegion;
            totalOffice = totalRegionsPlaceHolder.size();
        }
        System.out.println("Total office displayed on page is " + allOfficesDisplayed);
        return allOfficesDisplayed;
    }

    public void assertOffices(String test)
    {
        driver.findElements(By.xpath(".//*[@id='container']/section"));

    }

    public void assertTotalNumberOfOffices(int expectedOfficeDisplayed)
    {
        int returnedActualOfficesDisplayed = ReturnTotalNumberOfOfficesDisplay();
        Assert.assertEquals(returnedActualOfficesDisplayed,expectedOfficeDisplayed, "Expected "+expectedOfficeDisplayed+ "and found "+returnedActualOfficesDisplayed);
    }
